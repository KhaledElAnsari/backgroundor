Below the used code for this example 

      // javascript using jQuery
      $(document).ready(function() {
        $('header').backgroundor('lgradient',{intervaltime: 80, colors: ['#3b679e','#137FD8','#7db9e8']});
        $('#imgcontainer div').backgroundor('lgradient',{intervaltime: 70, colors: ['#7db9e8','#137FD8','#3b679e'], animdegree: '135deg'});
        $('#imgcontainer img').load(function() {
          setTimeout(function(){
            $("#imgcontainer div").fadeOut(1000,function(){
              $(this).remove();
            });
          },1500);
        });
      });
    
      /* css */
      #imgcontainer {
        position: relative;
        height: 300px;
        width: 400px;
        margin: 40px auto 0;
      }
      #imgcontainer img {
        width: 100%;
        height: 100%;
      }
      #imgcontainer div {
        position: absolute;
        top: 0;
        left: 0;
        width: 100%;
        height: 100%;
      }
