Below the used code for this example 

      // javascript using jQuery
      $(document).ready(function() {
        $('header').backgroundor('lgradient',{intervaltime: 80, colors: ['#3b679e','#137FD8','#7db9e8']});
        $('#pagepre').backgroundor('lgradient',{intervaltime: 70, colors: ['#7db9e8','#137FD8','#3b679e'], animdegree: '135deg'});
        $(window).load(function() {
          setTimeout(function(){
            $("#pagepre").fadeOut(1000,function(){
              $(this).remove();
            });
          },1500);
        });
      });
    
      /* css */
      #pagepre {
        position: fixed;
        width: 100%;
        height: 100%;
        top: 0;
        left: 0;
      }
